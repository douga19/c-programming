    #include <stdio.h>
#include <stdlib.h>

#define N_MARKS 5

int main (int argc, char const *argv[]) {
    double marks[N_MARKS] = {14, 12.5, 13, 16, 9.5};
    double total_marks = 0., average = 0.;

    for (int i = 0; i < N_MARKS; ++i) {
        total_marks += marks[i];
    }

    average = total_marks / N_MARKS;

    if (average < 10) {
        printf("FAILED\n");
    } else {
        printf("SUCCEEDED\n");
    }

    return EXIT_SUCCESS;
}
